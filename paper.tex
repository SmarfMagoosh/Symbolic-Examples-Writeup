\subsection{Syntax Definitions}
\par We may need rules for translation, dependency map building, eval, check, 
    scope checking, well-formedness checking.
\par We have rules for translation

\begin{enumerate}
    \item $\path$ is our path, or ``context.'' It keeps track of which labels are taking on which values currently.
    \item $\env$ is our environment. It functions as a dictionary that maps labels to values. Since a single variable may take
        on multiple values in different contexts, the environment really maps labels to dictionaries which map contexts to values.
    \item $\dmap$ is our dependency map. Since certain variable are only defined in the context of
        other variables, we need a map to tell us which other variable must be defined to bring another
        variable into scope. So the dependency maps labels to lists of labels for this purpose. For instance consider the
        symbolic expression $a -> (b -> b)*_{a}$. Since $b$ is defined in that star expression labelled ``$a$'' then
        $b$ can only be referenced if $a$ is defined in the current context of an expression. Thus, $\envget{\dmap}{b}=[a]$.
        When using the environment, we assume the dependency map is precomputed. We also assume the dependency map is consulted
        before looking up a variables specific value in the environment.
    \item $x \matches y$ means ``$x$ is matched by LRE $y$''
    \item $\envput{\lab}{x}$ means ``Insert the key value pair $(\lab, x)$ into the environment''
    \item $\envget{\lab}$ means ``lookup the value for $x$ in the environment.'' We assume that this operation takes the current
        context into account.
    \item $\envhas{E}{\lab}$ means ``check if $\lab$ is in the key set of $E$''
    \item Since all elements of a list need to be the same type, we will describe these types with the following system
        \begin{align*}
            Type = \base | [Type]
        \end{align*}
        We will use ``$\complex$'' to represent an arbitrary type.
    \item $\hastype{x}{T}$ means ``$x$ is of type $T$'' where $T$ is one of the aforementioned types.
    \item $x \translate y$ means ``$x$ translates to $y$''
\end{enumerate}

\subsection{Generation Rules}
\par The generation rules demonstrate how a nested list is generated from a symbolic expression
\begin{enumerate}
    \item $\inference
        [Dot]
        {}
        {\inp x \matches . \inp}$

    \item $\inference
        [VariableReference1]
        {\inp & \envhasnt{\env}{\lab} & (\envput{\env}{\lab}{x}, \path)}
        {\inp x \matches \lab (\envput{\env}{\lab}{x}, \path)}$
    \item $\inference
        [VariableReference2]
        {\inp & \envhas{\env}{\lab} & \envget{\env}{\lab} = x & \inp}
        {\inp x \matches \lab \inp}$

    \item $\inference
        [Empty]
        {}
        {\inp \nil \matches \nil \inp}$

    % by design, the depth of r_1 and r_2 are equal
    \item $\inference
        [UnionInit]
        {\inp & \envhasnt{\env}{\lab} & (\envput{\env}{\lab}{\gen}, \path) & x \matches r_1 |_{\lab} r_2 & \out}
        {\inp x \matches r_1 |_{\lab} r_2 \out}$
    \item $\inference
        [UnionLeft]
        {\inp & \envget{\env}{\lab} \mod 2 = 0 & (\env, \envput{\path}{\lab}{0}) & x \matches r_1 & \out}
        {\inp x \matches r_1 |_{\lab} r_2 \out}$
    \item $\inference
        [UnionRight]
        {\inp & \envget{\env}{\lab} \mod 2 = 1 & (\env, \envput{\path}{\lab}{1}) & x \matches r_2 & \out}
        {\inp x \matches r_1 |_{\lab} r_2 \out}$

    % by design, the starred regex has a non-zero depth
    \item $\inference
        [StarInit]
        {\inp & \envhasnt{\env}{\lab} & (\envput{\env}{\lab}{\gen}, \path) & x \matches r*_{\lab} & \out }
        {\inp x \matches r*_{\lab} \out}$
    \item $\inference
        [Star]
        {\inp & \envget{\env}{\lab} = n \inp xs \matches r\{n\} \out}
        {\inp xs \matches r*_{\lab} \dout}$
    \item $\inference
        [RepeatNBase]
        {n \leq 0}
        {\inp [] \matches r\{n\} \inp}$
    \item $\inference
        [RepeatN]
        {n > 0 & \inp & xs & \matches & r & \out & ys \matches r\{n-1\} & \dout}
        {\inp xs ++ ys \matches r\{n\} \inp}$

    % by design, r_1 and r_2 have a non-zero depth
    \item $\inference
        [Concat]
        {\inp xs \matches r_1 \out & ys \matches r_2 \dout}
        {\inp xs ++ ys \matches r_{1} :++ r_{2} \dout}$
    \item $\inference
        [Cons]
        {\inp x \matches r_1 \out xs \matches r_2 \dout}
        {\inp x : xs \matches r_{1} :> r_{2} \dout}$

    \item $\inference
        [Wrap]
        {\inp x \matches r \out}
        {\inp [x] \matches Wrap\;r \out}$

    \item $\inference
        [Literals]
        {}
        {\inp x \matches Lit\;x \out}$

    \item $\inference
        [ApplyBase]
        {}
        {\inp x \matches Data(x,\nil) \out}$
    \item $\inference
        [ApplyBase]
        {\inp x \matches Data(f(arg),args) \out}
        {\inp x \matches Data(f,(arg : args)) \out}$
\end{enumerate}

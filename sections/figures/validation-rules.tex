    % $\inference
    %     [Assignment]
    %     {\arbitrary{v} & \valinpwe{\lab}{v} x \gennedby e, \valout}
    %     {\valinp xs \gennedby (\lab \Longleftarrow \mathit{True}) \mapsto e, \valout}$
    $\inference
        [\ValidateAssignment]
        { args' = \forall 1 \leq i \leq n. \; (\env, \pathvar) arg_i \gens arg'_i, \env_i
        & (\env, \pathvar, \predtrack [(\pathvar, \lab) \rightarrow p(args')]) xs \gennedby e \gennedarrow ys, (\env', \predtrack'')}
        {\valinp xs \gennedby (\lab \Longleftarrow p(args, \lab)) \mapsto e \gennedarrow ys, (\env', \predtrack'')}$

    % list operations
    $\inference
        [\ValidateCons]
        {\valinp x \gennedby r_1 \gennedarrow [], \valout & \dinp xs \gennedby r_2 \gennedarrow ys, \dvalout}
        {\valinp x \ccons xs \gennedby r_1 \cons r_2 \gennedarrow ys, \dvalout}$
    $\inference
        [\ValidateConcat]
        {\valinp xs \gennedby r_1 \gennedarrow ys, \valout & \dinp ys \gennedby r_2 \gennedarrow zs, \dvalout}
        {\valinp xs  \gennedby r_1 \concat r_2 \gennedarrow zs, \dvalout}$
    $\inference
        [\ValidateEmpty]
        {}
        {\valinp ys \gennedby \nil \gennedarrow ys, \valinp}$

    % literals
    $\inference
        [\ValidateLiteral]
        {}
        {\valinp c \gennedby <c> \gennedarrow [], \valout}$
    $\inference
        [\ValidateApplication]
        {\env_0 = \env & \predtrack_0 = \predtrack
            & args' = \forall 0 \leq i < n. \; (\env_{i - 1}, \pathvar, \predtrack_{i - 1}) arg'_i \gennedby arg_i \gennedarrow \nil, (\env_i, \predtrack_i)}
         {\valinp <f \; args> \gens f(args') \gennedarrow \nil, (\env_n,\predtrack_n)}$

    % variables
    $\inference
        [\ValidateVarOne]
        {\envget{\lab} = x}
        {\valinp x \gennedby \lab \gennedarrow [], \valinp}$
    $\inference
        [\ValidateVarTwo]
        {\envhasnt{\lab} & (\predtrack_\pathvar(\lab))(x)}
        {\valinp x \gennedby \lab \gennedarrow [], (\outw{\lab}{x}, \predtrack)}$

    \vspace{0.5em}
    $\inference
        [\ValidateUnionSolveLeft]
        {
            \envhasnt{\lab}
        & (\predtrack_\pathvar(\lab))(0)
        \\ (\envput{\lab}{0}, \pathvar, \predtrack) xs \gennedby r_1 \gennedarrow ys, \valout
        }
        {\valinp xs \gennedby r_1 |_{\lab} r_2 \gennedarrow ys, \valout}$
    $\inference
        [\ValidateUnionSolveRight]
        {
            \envhasnt{\lab}
        & (\predtrack_\pathvar(\lab))(1)
        \\ (\envput{\lab}{1}, \pathvar, \predtrack) xs \gennedby r_2 \gennedarrow ys, \valout}
        {\valinp xs \gennedby r_1 |_{\lab} r_2 \gennedarrow ys, \valout}$
    $\inference
        [\ValidateUnionLeft]
        {\envget{\lab} \mod 2 = 0 & \valinp xs \gennedby r_1 \gennedarrow ys, \valout}
        {\valinp xs \gennedby r_1 |_{\lab} r_2 \gennedarrow ys, \valout}$
    $\inference
        [\ValidateUnionRight]
        {\envget{\lab} \mod 2 = 1 & \valinp xs \gennedby r_2 \gennedarrow ys, \valout}
        {\valinp xs \gennedby r_1 |_{\lab} r_2 \gennedarrow ys, \valout}$
    
    $\inference
        [\ValidateStarSolve]
        { \envhasnt{\lab}
            & \env_0 = \env
            & \text{choose } n \geq 0
            & \predtrack_\pathvar(\lab)(n)
            &\forall 0 \leq i < n . (\env_{i - 1}, \pathput{\lab}{i - 1}) ys_i \gennedby r \gennedarrow ys_{i}, (\env_{i}, \predtrack_i)
         }
        % {\envhasnt{\lab} & \exists n. \valinpwe{\lab}{n} xs \gennedby r*_{\lab}, \valout}
        {\valinp xs \gennedby r*_{lab} \gennedarrow ys_n, (\env_{n}, \predtrack_n)}$
    $\inference
         [\ValidateStar]
         { \envget{\lab} = n
            & \env_0 = \env & \predtrack_0 = \predtrack
            & \forall 0 \leq i < n . (\env_{i - 1}, \pathput{\lab}{i - 1}, \predtrack{i - 1}) ys_i \gennedby r \gennedarrow ys_{i}, (\env_{i}, \predtrack_i)
         }
         {\valinp xs \gennedby r*_{\lab} \gennedarrow ys_n, (\env_n, \predtrack_n)}$

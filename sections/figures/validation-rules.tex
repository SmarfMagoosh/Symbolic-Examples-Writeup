    % list operations
    $\inference
        [\ValidateCons]
        {\inp x \gennedby r_1 \gennedarrow [], \out & \dinp xs \gennedby r_2 \gennedarrow ys, \dout}
        {\inp x \ccons xs \gennedby r_1 \cons r_2 \gennedarrow ys, \dout}$
    $\inference
        [\ValidateConcat]
        {\inp xs \gennedby r_1 \gennedarrow ys, \out & \dinp ys \gennedby r_2 \gennedarrow zs, \dout}
        {\inp xs  \gennedby r_1 \concat r_2 \gennedarrow zs, \dout}$
    $\inference
        [\ValidateEmpty]
        {}
        {\inp ys \gennedby \nil \gennedarrow ys, \out}$

    % literals
    $\inference
        [\ValidateLiteral]
        {}
        {\inp c \gennedby <c> \gennedarrow [], \env}$
    $\inference
        [\ValidateApplication]
        {\env_0 = \env
            & args' = \forall 1 \leq i \leq n. \; (\env_{i - 1}, \pathvar) arg'_i \gennedby arg_i \gennedarrow \bill{...}, \env_i}
         {\inp <f \; args> \gens f(args') \gennedarrow \bill{...}, \out}$

    % variables
    $\inference
        [\ValidateVarOne]
        {\envget{\lab} = x}
        {\inp x \gennedby \lab \gennedarrow [], \out}$
    $\inference
        [\ValidateVarTwo]
        {\envhasnt{\lab} & \arbitrary{x}}
        {\inp x \gennedby \lab \gennedarrow [], \outw{\lab}{x}}$

    $\inference
        [\ValidateUnionSolveLeft]
        {\envhasnt{\lab} & \inp xs \gennedby r_1 \gennedarrow ys, \out}
        {\inp xs \gennedby r_1 |_{\lab} r_2 \gennedarrow ys, \outw{\lab}{0}}$
    $\inference
        [\ValidateUnionSolveRight]
        {\envhasnt{\lab} & \inp xs \gennedby r_2 \gennedarrow ys, \out}
        {\inp xs \gennedby r_1 |_{\lab} r_2 \gennedarrow ys, \outw{\lab}{1}}$
    $\inference
        [\ValidateUnionLeft]
        {\envget{\lab} \mod 2 = 0 & \inp xs \gennedby r_1 \gennedarrow ys, \out}
        {\inp xs \gennedby r_1 |_{\lab} r_2 \gennedarrow ys, \out}$
    $\inference
        [\ValidateUnionRight]
        {\envget{\lab} \mod 2 = 1 & \inp xs \gennedby r_2 \gennedarrow ys, \out}
        {\inp xs \gennedby r_1 |_{\lab} r_2 \gennedarrow ys, \out}$
    
    $\inference
        [\ValidateStarSolve]
        { \envhasnt{\lab}
            & \env_0 = \env
            & \text{choose } n \geq 0
            &\forall 1 \leq i < n . (\env_{i}, \pathput{\lab}{i}) ys_i \gennedby r \gennedarrow ys_{i + 1}, \env_{i + 1}
         }
        % {\envhasnt{\lab} & \exists n. \inpwe{\lab}{n} xs \gennedby r*_{\lab}, \out}
        {\inp xs \gennedby r*_{lab} \gennedarrow ys_n, \out}$
    $\inference
         [\ValidateStar]
         { \envget{\lab} = n
            & \env_0 = \env
            & \forall 1 \leq i < n . (\env_{i}, \pathput{\lab}{i}) ys_i \gennedby r \gennedarrow ys_{i + 1}, \env_{i + 1}
         }
         {\inp xs \gennedby r*_{\lab} \gennedarrow ys_n, \env_n}$

            &\OR \lab                                                  & \text{Label}    \\
            &\OR .                                                     & \text{Any Value}    \\
            &\OR (. \Longleftarrow b)                                  & \text{Constrained Value}    \\
            &\OR <x>                                                   & \text{Program Code}    \\ 
            &\OR \nil                                                  & \text{Empty List}    \\

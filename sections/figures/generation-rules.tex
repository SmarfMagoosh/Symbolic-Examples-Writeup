     $\inference
        [Variable]
        {\envget{\lab} = x}
        {\ginp \lab \gens x, \gout}$

     $\inference
        [Empty]
        {}
        {\ginp \nil \gens \nil, \gout}$

    % by design, the depth of r_1 and r_2 are equal
     $\inference
        [UnionLeft]
        {\envget{\lab} = 0 & \ginpwp{\lab}{0} r_1 \gens x, \gout}
        {\ginp r_1 |_{\lab} r_2 \gens x, \gout}$
     $\inference
        [UnionRight]
        {\envget{\lab} = 1 & \ginpwp{\lab}{0} r_2 \gens x, \gout}
        {\ginp r_1 |_{\lab} r_2 \gens x, \gout}$

    % Is it possible to pass the that label for the star when we switch it out for a constant label?
     $\inference
        [StarInit]
        {\envhasnt{\pathvar}{\lab} & \envget{\lab} = n & \ginpwp{\lab}{0} r*_{\lab} \gens xs, \gout}
        {\ginp r*_{\lab} \gens xs, \gout}$
     $\inference
        [Star]
        {\envget{\pathvar}{\lab} < n & \envget{\lab} = n & \ginp r \gens xs, \gout & \ginpwp{\lab}{\envget{\pathvar}{\lab} + 1} r*_{\lab} \gens ys}
        {\ginp r*_{\lab} \gens xs ++ ys}$
     $\inference
        [StarEnd]
        {\envget{\lab} \leq 0 \lor \envget{\pathvar}{\lab} \geq \envget{\lab}}
        {\ginp r*_{\lab} \gens \nil, \gout}$

    % by design, r_1 and r_2 have a non-zero depth for concat
     $\inference
        [Concat]
        {\ginp r_{1} \gens xs, \gout & r_{2} \gens ys, \gout}
        {\ginp r_{1} :++ r_{2} \gens xs ++ ys, \gout}$
     $\inference
        [Cons]
        {\ginp r_1 \gens x, \gout & \gout r_1 \gens xs, \gout}
        {\ginp r_{1} :> r_{2} \gens x : xs, \gout}$

        {}
        {\ginp<c> \gens c, \gout}$

        {\ginp <f \; args> \gens f(args'), \env}$

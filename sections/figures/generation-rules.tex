      % we assume all values are generated at assignment and therefore the environment contains them
      $\inference
         [\RuleVariable]
         {\envget{\lab} = x}
         {\inp \lab \gens x, \zout}$

      $\inference
         [\RuleEmpty]
         {}
         {\inp \nil \gens \nil, \zout}$

      % by design, the depth of r_1 and r_2 are equal
      $\inference
         [\RuleUnionLeft]
         {\envget{\lab} \mod 2 = 0 & \inp r_1 \gens x, \out}
         {\inp r_1 |_{\lab} r_2 \gens x, \out}$
      $\inference
         [\RuleUnionRight]
         {\envget{\lab} \mod 2 = 1 & \inp r_2 \gens x, \out}
         {\inp r_1 |_{\lab} r_2 \gens x, \out}$

      $\inference
         [\RuleStar]
         { \envget{\lab} = n
            & \env_0 = \env
            & xs = \forall 0 \leq i < n . (\env_{i}, \pathput{\lab}{i}) r \gens xs_{i + 1}, \env_{i + 1}
         }
         {\inp r*_{\lab} \gens \foldconcat{xs}, \env_{n}}$

      % by design, r_1 and r_2 have a non-zero depth for concat
      $\inference
         [\RuleConcat]
         {\inp r_1 \gens xs, \out & \dinp r_2 \gens ys, \dout}
         {\inp r_{1} \concat r_{2} \gens xs \append ys, \dout}$
      $\inference
         [\RuleCons]
         {\inp r_1 \gens x, \out & \dinp r_1 \gens xs, \dout}
         {\inp r_{1} \cons r_{2} \gens x \ccons xs, \dout}$

      % TODO be more clear about difference between this and apply maybe?
      $\inference
         [\RuleLiteral]
         {}
         {\inp <c> \gens c, \zout}$

      $\inference
         [\RuleApply]
         {\env_0 = \env
            & args' = \forall 1 \leq i \leq n. \; (\env_{i - 1}, \pathvar) arg_i \gens arg'_i, \env_i}
         {\inp <f \; args> \gens f(args'), \out}$

      $\inference
         [\RuleAssign]
         {\arbitrary{x} & p(args, x)}
         {\inp (\lab \Longleftarrow p(args, \lab)) \mapsto e, \outw{\lab}{x}}$
